Find Median from Data Stream

Streams pedem atualização incremental: duas heaps dividem os números em metade inferior e superior, mantendo a mediana nas raízes.

Raciocínio atual

Foco de entrevista: Declare os dois invariantes: esquerda ≤ direita e tamanhos diferem no máximo em 1.

Solução ingênua

Guarde todos os números, ordene tudo a cada inserção e leia o meio. É fácil, mas reprocessa o histórico inteiro em cada add.

  • addNum: O(n log n) se ordenar novamente.
  • findMedian: O(1) depois da lista ordenada.

Solução otimizada

Use max-heap para a metade inferior e min-heap para a metade superior. Insira, mova a raiz para preservar ordem entre metades e rebalanceie.

Invariante: lower guarda a metade menor, upper a metade maior, e lower pode ter no máximo um elemento a mais.

Big O

Cada inserção faz número constante de pushes/pops em heaps, logo O(log n). Ler uma ou duas raízes é O(1). O espaço é O(n).

Como falar em entrevista

  • Nomeie as heaps pelo papel, não pela implementação.
  • Mostre como calcular mediana quando tamanhos empatam ou não.
  • Aponte que todos os valores continuam armazenados.

Código de referência

class Heap {
  constructor(compare) { this.data = []; this.compare = compare; }
  peek() { return this.data[0]; }
  push(value) { this.data.push(value); this.up(this.data.length - 1); }
  pop() {
    const top = this.data[0];
    const last = this.data.pop();
    if (this.data.length) { this.data[0] = last; this.down(0); }
    return top;
  }
  up(i) { while (i > 0) { const p = (i - 1) >> 1; if (this.compare(this.data[p], this.data[i]) <= 0) break; [this.data[p], this.data[i]] = [this.data[i], this.data[p]]; i = p; } }
  down(i) { while (true) { let best = i; for (const c of [2*i+1, 2*i+2]) if (c < this.data.length && this.compare(this.data[best], this.data[c]) > 0) best = c; if (best === i) break; [this.data[i], this.data[best]] = [this.data[best], this.data[i]]; i = best; } }
}

class MedianFinder {
  constructor() { this.lower = new Heap((a, b) => b - a); this.upper = new Heap((a, b) => a - b); }
  addNum(num) {
    this.lower.push(num);
    this.upper.push(this.lower.pop());
    if (this.upper.data.length > this.lower.data.length) this.lower.push(this.upper.pop());
  }
  findMedian() {
    if (this.lower.data.length > this.upper.data.length) return this.lower.peek();
    return (this.lower.peek() + this.upper.peek()) / 2;
  }
}
type Compare<T> = (a: T, b: T) => number;
class Heap<T> {
  data: T[] = [];
  constructor(private compare: Compare<T>) {}
  peek(): T { return this.data[0]; }
  push(value: T): void { this.data.push(value); this.up(this.data.length - 1); }
  pop(): T { const top = this.data[0]; const last = this.data.pop()!; if (this.data.length) { this.data[0] = last; this.down(0); } return top; }
  private up(i: number): void { while (i > 0) { const p = (i - 1) >> 1; if (this.compare(this.data[p], this.data[i]) <= 0) break; [this.data[p], this.data[i]] = [this.data[i], this.data[p]]; i = p; } }
  private down(i: number): void { while (true) { let best = i; for (const c of [2*i+1, 2*i+2]) if (c < this.data.length && this.compare(this.data[best], this.data[c]) > 0) best = c; if (best === i) break; [this.data[i], this.data[best]] = [this.data[best], this.data[i]]; i = best; } }
}

class MedianFinder {
  private lower = new Heap<number>((a, b) => b - a);
  private upper = new Heap<number>((a, b) => a - b);
  addNum(num: number): void { this.lower.push(num); this.upper.push(this.lower.pop()); if (this.upper.data.length > this.lower.data.length) this.lower.push(this.upper.pop()); }
  findMedian(): number { return this.lower.data.length > this.upper.data.length ? this.lower.peek() : (this.lower.peek() + this.upper.peek()) / 2; }
}
