Top K Frequent Words

Raciocínio atual

Foco de entrevista: Explique que a heap guarda os k melhores e remove o pior entre eles, não o melhor global.
ingênuaO(n + m log m) tempo · O(m) espaço
otimizadaO(n + m log k + k log k) tempo · O(m + k) espaço

Solução ingênua

Conte frequências e ordene todas as palavras distintas pela regra oficial. É direto, mas ordena até candidatos que não entrarão no top k.

  • n: total de palavras; m: palavras distintas.
  • Ordenar todas as distintas custa O(m log m).

Solução otimizada

Conte as frequências e mantenha uma heap limitada a k candidatos. Ao final, ordene apenas os sobreviventes pela regra de saída.

Invariante: cada item na heap é melhor que qualquer candidato já descartado, e o topo representa o pior entre os mantidos.

Big O

A contagem custa O(n), a heap limitada custa O(m log k) e a ordenação final dos k itens custa O(k log k).

Como falar em entrevista

  • Comece pelo mapa de frequência.
  • Defina explicitamente a regra de desempate.
  • Diga por que ainda ordenamos os k finais antes de retornar.

class WordHeap {
  constructor(isWorse) {
    this.data = [];
    this.isWorse = isWorse;
  }

  push(item) {
    this.data.push(item);
    this.up(this.data.length - 1);
  }

  pop() {
    const top = this.data[0];
    const last = this.data.pop();
    if (this.data.length > 0) {
      this.data[0] = last;
      this.down(0);
    }
    return top;
  }

  up(index) {
    while (index > 0) {
      const parent = Math.floor((index - 1) / 2);
      if (!this.isWorse(this.data[index], this.data[parent])) break;
      [this.data[index], this.data[parent]] = [this.data[parent], this.data[index]];
      index = parent;
    }
  }

  down(index) {
    while (true) {
      let worst = index;
      for (const child of [2 * index + 1, 2 * index + 2]) {
        if (child < this.data.length && this.isWorse(this.data[child], this.data[worst])) {
          worst = child;
        }
      }
      if (worst === index) break;
      [this.data[index], this.data[worst]] = [this.data[worst], this.data[index]];
      index = worst;
    }
  }
}

function topKFrequent(words, k) {
  const freq = new Map();
  for (const word of words) freq.set(word, (freq.get(word) || 0) + 1);

  const isWorse = (a, b) => a.count === b.count ? a.word > b.word : a.count < b.count;
  const heap = new WordHeap(isWorse);

  for (const [word, count] of freq) {
    heap.push({ word, count });
    if (heap.data.length > k) heap.pop();
  }

  return heap.data
    .sort((a, b) => b.count - a.count || a.word.localeCompare(b.word))
    .map((entry) => entry.word);
}
type Entry = { word: string; count: number };

class WordHeap {
  data: Entry[] = [];
  constructor(private isWorse: (a: Entry, b: Entry) => boolean) {}

  push(item: Entry): void {
    this.data.push(item);
    this.up(this.data.length - 1);
  }

  pop(): Entry {
    const top = this.data[0];
    const last = this.data.pop()!;
    if (this.data.length > 0) {
      this.data[0] = last;
      this.down(0);
    }
    return top;
  }

  private up(index: number): void {
    while (index > 0) {
      const parent = Math.floor((index - 1) / 2);
      if (!this.isWorse(this.data[index], this.data[parent])) break;
      [this.data[index], this.data[parent]] = [this.data[parent], this.data[index]];
      index = parent;
    }
  }

  private down(index: number): void {
    while (true) {
      let worst = index;
      for (const child of [2 * index + 1, 2 * index + 2]) {
        if (child < this.data.length && this.isWorse(this.data[child], this.data[worst])) {
          worst = child;
        }
      }
      if (worst === index) break;
      [this.data[index], this.data[worst]] = [this.data[worst], this.data[index]];
      index = worst;
    }
  }
}

function topKFrequent(words: string[], k: number): string[] {
  const freq = new Map<string, number>();
  for (const word of words) freq.set(word, (freq.get(word) ?? 0) + 1);

  const isWorse = (a: Entry, b: Entry) => a.count === b.count ? a.word > b.word : a.count < b.count;
  const heap = new WordHeap(isWorse);

  for (const [word, count] of freq) {
    heap.push({ word, count });
    if (heap.data.length > k) heap.pop();
  }

  return heap.data
    .sort((a, b) => b.count - a.count || a.word.localeCompare(b.word))
    .map((entry) => entry.word);
}
