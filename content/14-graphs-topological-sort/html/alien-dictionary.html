Solução ingênua

Tentar permutações globais das letras vira tentativa e erro com C! ordens candidatas.

Solução otimizada

Use só palavras adjacentes: na primeira diferença a→b, adicione a aresta a antes de b; depois rode Kahn.

Big O

S é o total de caracteres; C letras distintas; E relações de precedência.

Como falar em entrevista

  • Trate prefixo inválido: “abc” antes de “ab”.
  • Inclua letras isoladas mesmo sem arestas.
  • Se Kahn não processar todas as letras, há ciclo.

function alienOrder(words) {
  const graph = new Map();
  const indegree = new Map();
  for (const word of words) {
    for (const ch of word) {
      if (!graph.has(ch)) graph.set(ch, new Set());
      if (!indegree.has(ch)) indegree.set(ch, 0);
    }
  }
  for (let i = 0; i < words.length - 1; i += 1) {
    const first = words[i], second = words[i + 1];
    if (first.length > second.length && first.startsWith(second)) return "";
    for (let j = 0; j < Math.min(first.length, second.length); j += 1) {
      if (first[j] !== second[j]) {
        if (!graph.get(first[j]).has(second[j])) {
          graph.get(first[j]).add(second[j]);
          indegree.set(second[j], indegree.get(second[j]) + 1);
        }
        break;
      }
    }
  }
  const queue = [...indegree.keys()].filter(ch => indegree.get(ch) === 0);
  const order = [];
  for (let head = 0; head < queue.length; head += 1) {
    const ch = queue[head];
    order.push(ch);
    for (const next of graph.get(ch)) {
      indegree.set(next, indegree.get(next) - 1);
      if (indegree.get(next) === 0) queue.push(next);
    }
  }
  return order.length === graph.size ? order.join("") : "";
}
function alienOrder(words: string[]): string {
  const graph = new Map<string, Set<string>>();
  const indegree = new Map<string, number>();
  for (const word of words) {
    for (const ch of word) {
      if (!graph.has(ch)) graph.set(ch, new Set<string>());
      if (!indegree.has(ch)) indegree.set(ch, 0);
    }
  }
  for (let i = 0; i < words.length - 1; i += 1) {
    const first = words[i], second = words[i + 1];
    if (first.length > second.length && first.startsWith(second)) return "";
    for (let j = 0; j < Math.min(first.length, second.length); j += 1) {
      if (first[j] !== second[j]) {
        if (!graph.get(first[j])!.has(second[j])) {
          graph.get(first[j])!.add(second[j]);
          indegree.set(second[j], indegree.get(second[j])! + 1);
        }
        break;
      }
    }
  }
  const queue = [...indegree.keys()].filter(ch => indegree.get(ch) === 0);
  const order: string[] = [];
  for (let head = 0; head < queue.length; head += 1) {
    const ch = queue[head];
    order.push(ch);
    for (const next of graph.get(ch)!) {
      indegree.set(next, indegree.get(next)! - 1);
      if (indegree.get(next) === 0) queue.push(next);
    }
  }
  return order.length === graph.size ? order.join("") : "";
}
